50  public class PartitionSelect extends SimpleBenchmark {
51    private static final int partitions = 10;
52    private static final int base = -358431684;
53    private static volatile int counter;
54    private static final ThreadLocal<Integer> tlsInt = new ThreadLocal<Integer>();
55    private static int randInt;
56    
57    @Override
58    protected void setUp() {
59      counter = 0;
60      tlsInt.set(new Integer(1));
61      randInt = (int) (Math.random() * 1000.0);
62    }
63    
64    public int timeModuloConst(int reps) {
65      int result = base;
66      for (int i = 0; i < reps; i++) {
67        result ^= result + randInt % partitions;
68      }
69      return result;
70    }
71    
72    public int timeModuloThreadId(int reps) {
73      int result = base;
74      for (int i = 0; i < reps; i++) {
75        result ^= result + Thread.currentThread().getId() % partitions;
76      }
77      return result;
78    }
79    
80    /*
81     * The AtomicInteger based tests are flawed in that the CASes are never
82     * contended. So the increments never see a CAS failure.
83     */
84    public int timeModuloAtomicCount(int reps) {
85      AtomicInteger counter = new AtomicInteger();
86      int result = base;
87      for (int i = 0; i < reps; i++) {
88        result ^= result + counter.incrementAndGet() % partitions;
89      }
90      return result;
91    }
92    
93    /*
94     * The Scalable Counter based tests are flawed in that there is never any
95     * contention. So the result is always accurate, and the cache-line never
96     * have to do a hand-over, so it will remain exclusive to the current CPU
97     * core. The x86 CPUs knows how to do this fast, presumably.
98     */
99    public int timeModuloScalableCounter(int reps) {
100     int result = base;
101     for (int i = 0; i < reps; i++) {
102       result ^= result + (++counter) % partitions;
103     }
104     return result;
105   }
106   
107   public int timeThreadLocalRef(int reps) {
108     int result = base;
109     for (int i = 0; i < reps; i++) {
110       result ^= result + tlsInt.get().intValue();
111     }
112     return result;
113   }
114   
115   public int timeModuloIdentityHashThread(int reps) {
116     int result = base;
117     for (int i = 0; i < reps; i++) {
118       result ^= result +
119           System.identityHashCode(Thread.currentThread()) % partitions;
120     }
121     return result;
122   }
123   
124   public int timeModuloIdentityHashNewObject(int reps) {
125     int result = base;
126     for (int i = 0; i < reps; i++) {
127       result ^= result + System.identityHashCode(new Object()) % partitions;
128     }
129     return result;
130   }
