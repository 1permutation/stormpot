16  package stormpot;
17  
18  import org.junit.Ignore;
19  import org.junit.Test;
20  
21  import java.lang.ref.WeakReference;
22  
23  import static org.junit.Assert.assertNull;
24  
25  public class ThreadLocalWeakRefTest {
26    /**
27     * Many Java web servers and containers go out of their way to find and null
28     * out ThreadLocals that an application might have created. Question is, is
29     * that really necessary?
30     */
31    @Ignore
32    @Test public void
33    threadsWeaklyReferenceThreadLocalValues() {
34      // Hypothesis: When we loose the reference to a ThreadLocal (e.g. by a
35      // web app being redeployed and getting a soft restart), and no other
36      // references to that object exists, then it should eventually be garbage
37      // collected.
38  
39      Object obj = new Object();
40      ThreadLocal<Object> threadLocal = new ThreadLocal<Object>();
41      WeakReference<Object> weakReference = new WeakReference<Object>(obj);
42      threadLocal.set(obj);
43  
44      // The stack of every thread is a GC-root. We have a strong reference to
45      // obj through the variable on our stack. When we clear it out, the only
46      // strong reference that will be left, will be through the threadLocal.
47      obj = null;
48  
49      // When we loose the reference to the threadLocal, then the values it
50      // referred to will no longer be strongly referenced by our thread.
51      threadLocal = null;
52  
53      // Our obj is now in principle garbage, though we still have a
54      // weakReference to it. The GC does not yet know this, because only the
55      // threadLocal object itself is now weakly referenced. We clear out that
56      // weak reference with a run of the GC.
57      System.gc();
58  
59      // The entry in our ThreadLocalMap is now stale. It still exists, and
60      // strongly references our obj, but it will be cleared next time our
61      // ThreadLocalMap does its clean-up procedure. We cannot invoke the
62      // clean-up procedure directly, but sufficient perturbation of the
63      // ThreadLocalMap will bring it about.
64      ThreadLocal<Object> a = new ThreadLocal<Object>();
65      a.set(new Object());
66      ThreadLocal<Object> b = new ThreadLocal<Object>();
67      b.set(new Object());
68  
69      // The entry has now been removed, and our obj is now weakly referenced
70      // by our weakReference above. Invoking another round of GC will clear it.
71      System.gc();
72  
73      // The obj has now been collected because all references are gone.
74      // We observe this by the weakReference being null.
75      assertNull(weakReference.get());
76  
77      // This test passes on Java 6, 7 and 8, with default GC settings.
78      // I haven't tried alternative GC settings, but you are welcome to do that.
79    }
80  }
