18  import static stormpot.UnitKit.*;
19  
20  import java.util.concurrent.Callable;
21  import java.util.concurrent.TimeUnit;
22  
23  import org.junit.Before;
24  import org.junit.Test;
25  
26  import stormpot.Config;
27  import stormpot.CountingAllocator;
28  import stormpot.Poolable;
29  
30  public class WhirlpoolTest {
31    Config config;
32    
33    @Before public void
34    setUp() {
35      config = new Config().setAllocator(new CountingAllocator());
36    }
37    
38    @Test(timeout = 300) public void
39    mustNotRemoveBlockedThreadsFromPublist() throws Exception {
40      // Github Issue #14
41      Whirlpool pool = givenDelayedReleaseToContendedPool(1);
42      // this must return before the test times out:
43      pool.claim();
44      pool.shutdown();
45    }
46  
47    private Whirlpool givenDelayedReleaseToContendedPool(
48        long claimTrafficTimeoutMs) throws InterruptedException {
49      config.setSize(1);
50      Whirlpool pool = new Whirlpool(config);
51      Poolable[] objs = new Poolable[] {pool.claim()};
52      Thread thread = fork($claimTrafficGenerator(pool, claimTrafficTimeoutMs));
53      fork($delayedReleases(objs, 20, TimeUnit.MILLISECONDS));
54      waitForThreadState(thread, Thread.State.RUNNABLE);
55      return pool;
56    }
57  
58    private Callable $claimTrafficGenerator(
59        final Whirlpool pool, final long timeout) {
60      return new Callable() {
61        public Object call() throws Exception {
62          try {
63            for (;;) {
64              // runs until interrupted
65              Poolable obj = pool.claim(timeout, TimeUnit.MILLISECONDS);
66              if (obj != null) {
67                obj.release();
68              }
69            }
70          } catch (Exception _) {
71            // ignore it
72          }
73          return null;
74        }
75      };
76    }
77    
78    @Test(timeout = 300) public void
79    blockedThreadsMustMakeProgressOverExpiredWaiters() throws Exception {
80      // Github Issue #15
81      Whirlpool pool = givenDelayedReleaseToContendedPool(-10);
82      // this must return before the test times out:
83      pool.claim(); // TODO dead-lock!
84      pool.shutdown();
85    }
