16  package stormpot.simulations;
17  
18  import stormpot.*;
19  
20  import java.util.concurrent.TimeUnit;
21  
22  /**
23   * A simulation that explores how the pool reacts to periods of low activity.
24   */
25  @Sim.Simulation(pools = {BlazePool.class}, measurementTime = 11, output = Sim.Output.summary)
26  public class VariableTrafficSim extends Sim {
27    private static final Timeout timeout = new Timeout(1, TimeUnit.MINUTES);
28    private volatile long startTime;
29  
30    @Conf(Param.expiration)
31    public TimeExpiration expiration = new TimeExpiration(6, TimeUnit.SECONDS);
32  
33    @Conf(Param.backgroundExpirationEnabled)
34    public boolean[] backgroundExpiration = {true, false};
35  
36    // We have a number of agents...
37    @Agents({@Agent, @Agent, @Agent, @Agent})
38    public void claimRelease(Pool<Poolable> pool) throws InterruptedException {
39      if (startTime == 0) {
40        startTime = System.currentTimeMillis();
41      }
42      pool.claim(timeout).release();
43    }
44  
45    // ... that for an initial period are very busy ...
46    @AgentPause(unit = TimeUnit.MILLISECONDS)
47    public long pause() {
48      long now = System.currentTimeMillis() - startTime;
49      if (now < 5000) {
50        return 100; // short pause for the first 5 seconds
51      }
52  
53      // ... then suddenly they sleep for 5 seconds ...
54      if (now < 6000) {
55        // ... while they sleep, all the objects will expire at the 6 second point ...
56        return 5000;
57        // ... this leaves 3 seconds for any background allocation to replace the expired objects ...
58      }
59  
60      // ... and then they suddenly become very active again ...
61      return 100;
62    }
63  
64    // ... if no expiration checks happen in the background ...
65    @AllocationCost
66    public long allocationCost() {
67      // ... then this high allocation cost will hit the agents hard, when they wake up.
68      return 500;
69      // Each object costs 500 milliseconds to allocate, so allocating enough objects to
70      // satisfy the agents will take 2 seconds.
71    }
72  }
