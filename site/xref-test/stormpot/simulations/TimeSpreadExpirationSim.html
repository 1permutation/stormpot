18  import stormpot.*;

20  import java.util.concurrent.TimeUnit;

22  /**
23   * This simulation examines the difference that using TimeSpreadExpiration,
24   * instead of TimeExpiration, can have on latency.
25   */
26  @Sim.Simulation(pools = {BlazePool.class}, measurementTime = 60, output = Sim.Output.summary)
27  public class TimeSpreadExpirationSim extends Sim {
28    @Conf(Param.expiration)
29    public Expiration[] expirations = {
30        new TimeExpiration(2, TimeUnit.SECONDS),
31        new TimeSpreadExpiration(1, 2, TimeUnit.SECONDS)
32    };
33  
34    @Agents({@Agent, @Agent, @Agent, @Agent, @Agent, @Agent, @Agent})
35    public void claimRelease(Pool<GenericPoolable> pool) throws InterruptedException {
36      pool.claim(new Timeout(1, TimeUnit.MINUTES)).release();
37    }
38  
39    @AgentPause
40    public long pause() {
41      return 100;
42    }
43  
44    @AllocationCost
45    public long allocationCost() {
46      return 100;
47    }
48  }
