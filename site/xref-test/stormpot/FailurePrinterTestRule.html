16  package stormpot;
17  
18  import org.junit.rules.TestRule;
19  import org.junit.runner.Description;
20  import org.junit.runners.model.Statement;
21  
22  import java.lang.management.*;
23  import java.util.List;
24  
25  /**
26   * A TestRule that ensures that any failing tests have their stack-trace
27   * printed to stderr.
28   * 
29   * This is useful for when the tests are running on a build-server, and you'd
30   * like the details of any failure to be printed to the build-log. A nice thing
31   * if the build system does not make the build artifacts with the test failures
32   * available.
33   * 
34   * @author cvh
35   */
36  public class FailurePrinterTestRule implements TestRule {
37    @Override
38    public Statement apply(final Statement base, final Description description) {
39      return new Statement() {
40        @Override
41        public void evaluate() throws Throwable {
42          ThreadMXBean threadMXBean = ManagementFactory.getThreadMXBean();
43          CompilationMXBean compilationMXBean = ManagementFactory.getCompilationMXBean();
44          List<GarbageCollectorMXBean> garbageCollectorMXBeans = ManagementFactory.getGarbageCollectorMXBeans();
45          OperatingSystemMXBean operatingSystemMXBean = ManagementFactory.getOperatingSystemMXBean();
46  
47          long startCpuTimeNs = getCpuTimeNs(threadMXBean);
48          long startUserTimeNs = getUserTimeNs(threadMXBean);
49          int startThreadCount = threadMXBean.getThreadCount();
50          long startCompilationTimeMs = getCompilationTimeMs(compilationMXBean);
51          long startCollectionTimeMs = collectionTimeMs(garbageCollectorMXBeans);
52          double loadAverage = operatingSystemMXBean.getSystemLoadAverage();
53  
54          int cores = operatingSystemMXBean.getAvailableProcessors();
55  
56          try {
57            base.evaluate();
58          } catch (Throwable th) {
59            long cpuTimeNs = getCpuTimeNs(threadMXBean) - startCpuTimeNs;
60            long userTimeNs = getUserTimeNs(threadMXBean) - startUserTimeNs;
61            int threadCount = threadMXBean.getThreadCount();
62            long compilationTimeMs = getCompilationTimeMs(compilationMXBean) - startCompilationTimeMs;
63            long collectionTimeMs = collectionTimeMs(garbageCollectorMXBeans) - startCollectionTimeMs;
64            double newLoadAverage = operatingSystemMXBean.getSystemLoadAverage();
65            System.err.printf("Failed test:%n" +
66                "\tcpu-cores = %s%n" +
67                "\tcpuTimeNs = %s%n" +
68                "\tuserTimeNs = %s%n" +
69                "\tcompilationTimeMs = %s%n" +
70                "\tcollectionTimeMs = %s%n" +
71                "\tload average = %s -> %s%n" +
72                "\tthreads = %s -> %s%n",
73                cores,
74                cpuTimeNs,
75                userTimeNs,
76                compilationTimeMs,
77                collectionTimeMs,
78                loadAverage, newLoadAverage,
79                startThreadCount, threadCount);
80            th.printStackTrace();
81            System.err.flush();
82            throw th;
83          }
84        }
85      };
86    }
87  
88    private long collectionTimeMs(List<GarbageCollectorMXBean> garbageCollectorMXBeans) {
89      long sum = 0L;
90      for (GarbageCollectorMXBean gc : garbageCollectorMXBeans) {
91        long collectionTime = gc.getCollectionTime();
92        if (collectionTime != -1) {
93          sum += collectionTime;
94        }
95      }
96      return sum;
97    }
98  
99    private long getCompilationTimeMs(CompilationMXBean compilationMXBean) {
100     return compilationMXBean.isCompilationTimeMonitoringSupported()?
101         compilationMXBean.getTotalCompilationTime() : 0;
102   }
103 
104   private long getUserTimeNs(ThreadMXBean threadMXBean) {
105     return threadMXBean.isCurrentThreadCpuTimeSupported()?
106         threadMXBean.getCurrentThreadUserTime() : 0;
107   }
108 
109   private long getCpuTimeNs(ThreadMXBean threadMXBean) {
110     return threadMXBean.isCurrentThreadCpuTimeSupported()?
111         threadMXBean.getCurrentThreadCpuTime() : 0;
112   }
113 }
