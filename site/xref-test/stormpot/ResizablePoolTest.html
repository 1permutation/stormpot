83    /**
84     * When we increase the size of a depleted pool, it should be possible to
85     * make claim again and get out newly allocated objects.
86     * 
87     * We test for this by depleting a pool, upping the size and then claiming
88     * again with a timeout that is longer than the timeout of the test. The test
89     * pass if it does not timeout.
90     * @param fixture
91     * @throws Exception
92     */
93    @Test(timeout = 300)
94    @Theory public void
95    increasingSizeMustAllowMoreAllocations(PoolFixture fixture) throws Exception {
96      ResizablePool<GenericPoolable> pool = resizable(fixture);
97      pool.claim(longTimeout); // depleted
98      pool.setTargetSize(2);
99      // now this mustn't block:
100     pool.claim(longTimeout);
101   }

103   /**
104    * We must somehow ensure that the pool starts deallocating more than it
105    * allocates, when the pool is shrunk. This is difficult because the pool
106    * cannot tell us when it reaches the target size, so we have to figure this
107    * out by using a special allocator.
108    * 
109    * We test for this by configuring a CountingAllocator that also unparks a
110    * thread (namely ours, the main thread for the test) at every allocation
111    * and deallocation. We also configure the pool to have a somewhat large
112    * initial size, so we can shrink it later. Then we deplete the pool, and
113    * set a smaller target size. After setting the new target size, we release
114    * just enough objects for the pool to reach it, and then we wait for the
115    * allocator to register that same number of deallocations. This has to
116    * happen before the test times out. After that, we check that the difference
117    * between the allocations and the deallocations matches the new target size.
118    * @param fixture
119    * @throws Exception
120    */
121   @Test(timeout = 300)
122   @Theory public void
123   decreasingSizeMustEventuallyDeallocateSurplusObjects(PoolFixture fixture)
124       throws Exception {
125     int startingSize = 5;
126     int newSize = 1;
127     final Thread main = Thread.currentThread();
128     CountingAllocator allocator = new UnparkingCountingAllocator(main);
129     config.setSize(startingSize);
130     config.setAllocator(allocator);
131     ResizablePool<GenericPoolable> pool = resizable(fixture);
132     List<GenericPoolable> objs = new ArrayList<GenericPoolable>();
133     
134     while (allocator.allocations() != startingSize) {
135       objs.add(pool.claim(longTimeout)); // force the pool to do work
136     }
137     pool.setTargetSize(newSize);
138     while (allocator.deallocations() != startingSize - newSize) {
139       if (objs.size() > 0) {
140         objs.remove(0).release(); // give the pool objects to deallocate
141       } else {
142         pool.claim(longTimeout).release(); // prod it & poke it
143       }
144       LockSupport.parkNanos(10000000); // 10 millis
145     }
146     assertThat(
147         allocator.allocations() - allocator.deallocations(), is(newSize));
148   }

150   /**
151    * Similar to the decreasingSizeMustEventuallyDeallocateSurplusObjects test
152    * above, but this time the objects have a very short TTL and we wait with
153    * releasing them until they have expired.
154    * 
155    * Again, we deplete the pool and then spinwait for the objects to expire.
156    * Then we set the new lower target size, and release just enough for the
157    * pool to reach the new target. Then we try to claim an object from the pool
158    * with a very short timeout. This will return null because the pool is still
159    * depleted. We also check that the pool has not made any new allocations,
160    * even though we have been releasing objects. We don't check the
161    * deallocations because it's complicated and we did it in the
162    * decreasingSizeMustEventuallyDeallocateSurplusObjects test above.
163    * @param fixture
164    * @throws Exception
165    */
166   @Test(timeout = 300)
167   @Theory public void
168   mustNotReallocateWhenReleasingExpiredObjectsIntoShrunkPool(PoolFixture fixture)
169       throws Exception {
170     int startingSize = 5;
171     int newSize = 1;
172     CountingAllocator allocator = new CountingAllocator();
173     Expiration<Poolable> expiration =
174         new TimeExpiration(1, TimeUnit.MILLISECONDS);
175     config.setExpiration(expiration).setAllocator(allocator);
176     config.setSize(startingSize);
177     ResizablePool<GenericPoolable> pool = resizable(fixture);
178     List<GenericPoolable> objs = new ArrayList<GenericPoolable>();
179     while (allocator.allocations() < startingSize) {
180       objs.add(pool.claim(longTimeout));
181     }
182     UnitKit.spinwait(2); // wait for the objects to expire
183     pool.setTargetSize(newSize);
184     for (int i = 0; i < startingSize - newSize; i++) {
185       // release the surplus expired objects back into the pool
186       objs.remove(0).release();
187     }
188     // now the released objects should not cause reallocations, so claim
189     // returns null (it's still depleted) and allocation count stays put
190     assertThat(pool.claim(shortTimeout), nullValue());
191     assertThat(allocator.allocations(), is(startingSize));
192   } // TODO racy?? I don't know... :(
