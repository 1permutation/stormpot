49  /**
50   * This is the generic test for Pool implementations. The test ensures that
51   * an implementation adheres to the general contract of the Pool interface,
52   * given certain circumstances and standardised configurations.
53   * <p>
54   * Pools may have other properties, and may be configurable to deviate from
55   * the standardised behaviour. However, such properties must not be
56   * observable within the standardised spectrum of configurations.
57   * <p>
58   * The tests for any special properties that a pool may have, must be put in
59   * a pool-specific test case. Do not use assumptions or other tricks to
60   * pollute this test case with tests for pool-specific or non-standard
61   * behaviours and configurations.
62   * <p>
63   * The test case uses theories to apply to the set of possible Pool
64   * implementations. Each implementation must have a PoolFixture, which is
65   * used to construct and initialise the pool, based on a {@link Config}.
66   * <p>
67   * The only assumptions used in this test, is whether the Pool is a
68   * LifecycledPool or not. And most interesting pools are life-cycled.
69   * LifecycledPools can be shut down. This is a required ability, in order to
70   * test a number of behaviours, but also brings about its own set of new
71   * behaviours and flows that needs to be tested for. Those tests are also
72   * included in this test case.
73   * <p>
74   * <strong>Note:</strong> when adding, removing or modifying tests, also
75   * remember to update the {@link Pool} javadoc - especially the part about
76   * what promises are provided by the Pool interface and its implementations.
77   *

79   * @see Pool

453   /**
454    * SlotInfo instances must have a means of acting as a non-contended source
455    * of random numbers. We test this by getting a hold of a SlotInfo instance,
456    * and then pulling a large quantity of random numbers from it. If the
457    * numbers are random, then they will have a roughly even split between ones
458    * and zero bits.
459    */
460   @Test(timeout = 1601)
461   @Theory public void
462   slotInfoMustBeAbleToProduceRandomNumbers(PoolFixture fixture)
463       throws Exception {
464     final AtomicReference<SlotInfo<? extends Poolable>> slotInfoRef =
465         new AtomicReference<SlotInfo<? extends Poolable>>();
466     config.setExpiration(expire($capture($slotInfo(slotInfoRef), $fresh)));
467     createPool(fixture);
468     pool.claim(longTimeout).release(); // Now we have a SlotInfo reference.

1738   /**
1739    * Here's the scenario we're trying to target:
1740    *
1741    * - You (or your thread) do a successful claim, and triumphantly stores it
1742    *   in the ThreadLocal cache.
1743    * - You then return the object after use, so now it's back in the
1744    *   live-queue for others to grab.
1745    * - Someone else tries to claim the object, but decides that it has expired,
1746    *   and sends it off through the dead-queue to be reallocated.
1747    * - The reallocation fails for some reason, and the slot is now poisoned.
1748    * - You want to claim an object again, and start by looking in the
1749    *   ThreadLocal cache.
1750    * - You find the slot for the object you had last, but the slot is poisoned.
1751    * - Now, because you found it in the ThreadLocal cache – and notably did
1752    *   *not* pull it off of the live-queue – you cannot just put it on the
1753    *   dead-queue, because that could lead to unbounded memory use.
1754    * - Instead, it has to be marked as live, and we instead have to wait for
1755    *   someone to pull it off of the live-queue, check the poison again, and
1756    *   *then* put it on the dead-queue.
1757    * - Your ThreadLocal reclaim attempt then end in throwing the poison,
1758    *   wrapped in a PoolException.
1759    * - Sadly, this process does not involve clearing out the ThreadLocal cache,
1760    *   so if you quickly catch the exception and try to claim again, you will
1761    *   find the same exact poisoned slot and go through the same routine, that
1762    *   ends in a thrown exception and a poisoned slot still left in the
1763    *   ThreadLocal cache.
1764    */
1765   @Test(timeout=601)
1766   @Theory public void
1767   mustNotCachePoisonedSlots(PoolFixture fixture) throws Exception {
1768     // First we prime the possible thread-local cache to a particular object.
1769     // Then we instruct the allocator to always throw an exception when it is
1770     // told to allocate on that particular slot.
1771     // Then, in another thread, we mark all objects in the pool as expired.
1772     // Once we have observed a reallocation attempt at our primed slot, we
1773     // try to reclaim it. The reclaim must not throw an exception because of
1774     // the cached poisoned slot.
1775     config.setSize(3);
1776 
1777     // Enough permits for each initial allocation:
1778     final Semaphore semaphore = new Semaphore(3);
1779 
1780     final AtomicBoolean hasExpired = new AtomicBoolean(false);
1781     config.setExpiration(expire($expiredIf(hasExpired)));
1782 
1783     final String allocationCause = "allocation blew up!";
1784     final AtomicReference<Slot> failOnAllocatingSlot =
1785         new AtomicReference<Slot>();
1786     final AtomicInteger observedFailedAllocation = new AtomicInteger();
1787     Action observeFailure = new Action() {
1788       @Override
1789       public GenericPoolable apply(Slot slot, GenericPoolable obj) throws Exception {
1790         if (slot == failOnAllocatingSlot.get()) {
1791           observedFailedAllocation.incrementAndGet();
1792           throw new RuntimeException(allocationCause);
1793         }
1794         return new GenericPoolable(slot);
1795       }
1796     };
1797     allocator = allocator(alloc($acquire(semaphore, observeFailure)));
1798     config.setAllocator(allocator);
1799 
1800     createPool(fixture);
1801 
1802     // Wait for the pool to fill
1803     while (allocator.countAllocations() < 3) {
1804       Thread.yield();
1805     }
1806 
1807     // Prime any thread-local cache
1808     GenericPoolable obj = pool.claim(longTimeout);
1809     failOnAllocatingSlot.set(obj.slot);
1810     obj.release(); // Places slot at end of queue
1811
