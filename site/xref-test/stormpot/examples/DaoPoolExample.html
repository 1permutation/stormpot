32  public class DaoPoolExample {
33    static class MyDao implements Poolable {
34      private final Slot slot;
35      private final Connection connection;
36      
37      private MyDao(Slot slot, Connection connection) {
38        this.slot = slot;
39        this.connection = connection;
40      }
41  
42      public void release() {
43        slot.release(this);
44      }
45  
46      private void close() throws SQLException {
47        connection.close();
48      }
49      
50      // ...
51      // public DAO-type methods
52      // ...
53      
54      public String getFirstName() {
55        // Stub: get the name from the database using the connection.
56        // But for now, just always return "freddy"
57        return "freddy";
58      }
59    }
60    
61    static class MyDaoAllocator implements Allocator<MyDao> {
62      private final DataSource dataSource;
63      
64      public MyDaoAllocator(DataSource dataSource) {
65        this.dataSource = dataSource;
66      }
67  
68      public MyDao allocate(Slot slot) throws Exception {
69        synchronized (dataSource) {
70          return new MyDao(slot, dataSource.getConnection());
71        }
72      }
73  
74      public void deallocate(MyDao poolable) throws Exception {
75        poolable.close();
76      }
77    }
78    
79    static class MyDaoPool {
80      private final LifecycledPool<MyDao> pool;
81      
82      public MyDaoPool(DataSource dataSource) {
83        MyDaoAllocator allocator = new MyDaoAllocator(dataSource);
84        Config<MyDao> config = new Config<MyDao>().setAllocator(allocator);
85        pool = new QueuePool<MyDao>(config);
86      }
87  
88      public void close() throws InterruptedException {
89        pool.shutdown().await(new Timeout(1, TimeUnit.MINUTES));
90      }
91      
92      public <T> T doWithDao(WithMyDaoDo<T> action)
93      throws InterruptedException {
94        MyDao dao = pool.claim(new Timeout(1, TimeUnit.SECONDS));
95        try {
96          return action.doWithDao(dao);
97        } finally {
98          dao.release();
99        }
100     }
101   }
102   
103   static interface WithMyDaoDo<T> {
104     public T doWithDao(MyDao dao);
105   }
106   
107   public static void main(String[] args) throws InterruptedException {
108     DataSource dataSource = configureDataSource();
109     MyDaoPool pool = new MyDaoPool(dataSource);
110     try {
111       String person = pool.doWithDao(new WithMyDaoDo<String>() {
112         public String doWithDao(MyDao dao) {
113           return dao.getFirstName();
114         }
115       });
116       System.out.println("Hello there, " + person + "!");
117     } finally {
118       pool.close();
119     }
120   }
121

127   }
