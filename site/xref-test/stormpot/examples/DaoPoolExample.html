87    static class TestQueryExpiration implements Expiration<MyDao> {
88      @Override
89      public boolean hasExpired(SlotInfo<? extends MyDao> info) {
90        MyDao dao = info.getPoolable();
91        Connection con = dao.connection;
92        Statement stmt = null;
93        synchronized (con) {
94          try {
95            try {
96              stmt = con.createStatement();
97              stmt.execute("select 1 from dual;");
98            } finally {
99              if (stmt != null) {
100               stmt.close();
101             }
102           }
103         } catch (SQLException e) {
104           return true;
105         }
106         return false;
107       }

111   // tag::poolStart[]
112   static class MyDaoPool {
113     private final LifecycledPool<MyDao> pool;
114     
115     public MyDaoPool(DataSource dataSource) {
116       MyDaoAllocator allocator = new MyDaoAllocator(dataSource);
117       Config<MyDao> config = new Config<MyDao>().setAllocator(allocator);
118       pool = new QueuePool<MyDao>(config);
119     }
120     // end::poolStart[]
121 
122     public MyDaoPool(DataSource dataSource, boolean verifyConnections) {
123       MyDaoAllocator allocator = new MyDaoAllocator(dataSource);
124       Config<MyDao> config = new Config<MyDao>().setAllocator(allocator);
125 
126       if (verifyConnections) {
127         config.setExpiration(new TestQueryExpiration());
128       }
129 
130       pool = new QueuePool<MyDao>(config);
131     }
132 
133     // tag::poolClose[]
134     public void close() throws InterruptedException {
135       pool.shutdown().await(new Timeout(1, TimeUnit.MINUTES));
136     }
137     // end::poolClose[]

139     // tag::poolDoWithDao[]
140     public <T> T doWithDao(WithMyDaoDo<T> action)
141         throws InterruptedException {
142       MyDao dao = pool.claim(new Timeout(1, TimeUnit.SECONDS));
143       try {
144         return action.doWithDao(dao);
145       } finally {
146         dao.release();
147       }
148     }
