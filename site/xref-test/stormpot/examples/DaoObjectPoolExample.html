66      
67      public MyDaoObjectPool(DataSource dataSource) {
68        MyDaoObjectAllocator allocator = new MyDaoObjectAllocator(dataSource);
69        Config<MyDaoObject> config = new Config().setAllocator(allocator);
70        pool = new QueuePool(config);
71      }
72      
73      public <T> T doWithDao(WithMyDaoObjectDo<T> action)
74      throws InterruptedException {
75        MyDaoObject dao = pool.claim();
76        try {
77          return action.doWithDao(dao);
78        } finally {
79          dao.release();
80        }
81      }

83      public void close() throws InterruptedException {

85      }
