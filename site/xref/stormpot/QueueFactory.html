16  package stormpot;
17  
18  import java.lang.reflect.Constructor;
19  import java.util.concurrent.BlockingQueue;
20  import java.util.concurrent.LinkedBlockingQueue;
21  
22  class QueueFactory {
23    interface Factory {
24      <T> BlockingQueue<T> createUnboundedBlockingQueue();
25    }
26  
27    private static final Factory factory;
28  
29    static {
30      Factory theFactory = null;
31      String queueClass = System.getProperty(
32          "stormpot.blocking.queue.impl",
33          "java.util.concurrent.LinkedTransferQueue");
34      try {
35        Class<?> transferqueueClass = Class.forName(queueClass);
36        final Constructor<?> constructor = transferqueueClass.getConstructor();
37        constructor.newInstance(); // Just making sure that it works...
38        theFactory = new Factory() {
39          @SuppressWarnings("unchecked")
40          @Override
41          public <T> BlockingQueue<T> createUnboundedBlockingQueue() {
42            try {
43              return (BlockingQueue<T>) constructor.newInstance();
44            } catch (Exception e) {
45              Exception exception = new Exception(
46                  "WARNING: Expected class to be constructable: " + constructor,
47                  e);
48              exception.printStackTrace();
49            }
50            return new LinkedBlockingQueue<T>();
51          }
52        };
53      } catch (Exception ignore) {
54      }
55      if (theFactory == null) {
56        theFactory = new Factory() {
57          @Override
58          public <T> BlockingQueue<T> createUnboundedBlockingQueue() {
59            return new LinkedBlockingQueue<T>();
60          }
61        };
62      }
63      factory = theFactory;
64    }
65  
66    static <T> BlockingQueue<T> createUnboundedBlockingQueue() {
67      return factory.createUnboundedBlockingQueue();
68    }
69  }
