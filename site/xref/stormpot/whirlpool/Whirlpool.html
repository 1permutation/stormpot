32  /**
33   * Whirlpool is a {@link Pool} that is based on a queue-like structure, made
34   * concurrent using the Flat-Combining technique of Hendler, Incze, Shavit and
35   * Tzafrir.
36   * <p>
37   * <strong>NOTE:</strong> This Pool implementation is still experimental.
38   * Use {@link QueuePool} for an implementation that is currently considered
39   * more reliable.
40   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
41   * @param <T> The type of {@link Poolable} managed by this pool.
42   *
43   */
44  public class Whirlpool<T extends Poolable> implements LifecycledPool<T> {
45    static final int LOCKED = 1;
46    static final int UNLOCKED = 0;
47    static final int WAIT_SPINS = 128;
48    static final int CLEANUP_MASK = (1 << 12) - 1;
49    static final int PARK_TIME_NS = 1000000;
50    static final int EXPIRE_PASS_COUNT = 100;
51  
52    static final WSlot CLAIM = new WSlot(null, "claim");
53    static final WSlot RELIEVE = new WSlot(null, "relieve");
54    static final WSlot RELEASE = new WSlot(null, "release");
55    static final WSlot SHUTDOWN = new WSlot(null, "shutdown");
56    static final WSlot TIMEOUT = new WSlot(null, "timeout");
57    static final WSlot INTERRUPT = new WSlot(null, "interrupt");
58    
59    static final AtomicReferenceFieldUpdater<Whirlpool, Request> publistCas =
60      newUpdater(Whirlpool.class, Request.class, "publist");
61    
62    private final RequestThreadLocal requestTL = new RequestThreadLocal();
63    
64    private volatile Request publist;
65    private final AtomicInteger lock = new AtomicInteger(UNLOCKED);
66    private volatile boolean shutdown = false;
67    private int combiningPass;
68    private WSlot liveStack;
69    private WSlot deadStack;
70    private long ttl;
71    private WpAllocThread alloc;

73    /**
74     * Construct a new Whirlpool instance from the given {@link Config}.
75     * @param config The pool configuration to use.
76     */
77    public Whirlpool(Config<T> config) {
78      synchronized (config) {
79        config.validate();
80        ttl = config.getTTLUnit().toMillis(config.getTTL());
81        alloc = new WpAllocThread(config, this);
82      }
83      alloc.start();
84    }

86    WSlot relieve(long timeout, TimeUnit unit) throws InterruptedException {
87      Request request = getPrepareRequest();
88      request.setTimeout(timeout, unit);
89      request.requestOp = RELIEVE;
90      return perform(request);
91    }
92  
93    private Request getPrepareRequest() {
94      Request request = requestTL.get();
95      if (request.requestOp != null) {
96        throw new AssertionError(
97            "requesting thread have stale request: " + request.requestOp);
98      }
99      if (request.response != null) {
100       throw new AssertionError(
101           "requesting thread have stale response: " + request.response);
102     }
103     return request;
104   }

106   public T claim() throws PoolException, InterruptedException {
107     preClaimCheck();
108     Request request = getPrepareRequest();
109     request.setNoTimeout();
110     request.requestOp = CLAIM;
111     WSlot slot = perform(request);
112     return objectOf(slot);
113   }
114 
115   private void preClaimCheck() throws InterruptedException {
116     if (shutdown) {
117       throw new IllegalStateException("pool is shut down");
118     }
119     if (Thread.interrupted()) {
120       throw new InterruptedException();
121     }
122   }
123   
124   public T claim(long timeout, TimeUnit unit) throws PoolException,
125       InterruptedException {
126     if (unit == null) {
127       throw new IllegalArgumentException("timeout TimeUnit cannot be null.");
128     }
129     preClaimCheck();
130     Request request = getPrepareRequest();
131     request.setTimeout(timeout, unit);
132     request.requestOp = CLAIM;
133     WSlot slot = perform(request);
134     return objectOf(slot);
135   }

137   @SuppressWarnings("unchecked")
138   private T objectOf(WSlot slot) {
139     if (slot == null) {
140       return null;
141     }
142     if (slot.poison != null) {
143       Exception exception = slot.poison;
144       slot.created = 0;
145       release(slot);
146       throw new PoolException("allocation failed", exception);
147     }
148     slot.claimed = true;
149     return (T) slot.obj;
150   }

152   void release(WSlot slot) {
153     Request request = getPrepareRequest();
154     request.setTimeout(1, TimeUnit.HOURS);
155     request.requestOp = slot;
156     try {
157       perform(request);
158     } catch (InterruptedException e) {
159       // this is not possible, but regardless...
160       Thread.currentThread().interrupt();
161     }
162   }
163 
164   private WSlot perform(Request request)
165   throws InterruptedException {
166     for (;;) {
167       if (request.active) {
168         // step 2
169         if (lock.compareAndSet(UNLOCKED, LOCKED)) { // step 3
170           // step 4 - got lock - we are now a combiner
171           combiningPass++;
172           scanCombineApply();
173           if ((combiningPass & CLEANUP_MASK) == CLEANUP_MASK) {
174 //            cleanUp();
175           }
176           lock.set(UNLOCKED);
177           WSlot slot = request.response;
178           if (slot == null) {
179             request.await();
180             continue;
181           }
182           return replyOf(request);
183         } else {
184           // step 2 - did not get lock - spin-wait for response
185           for (int i = 0; i < WAIT_SPINS; i++) {
186             WSlot slot = request.response;
187             if (slot != null) {
188               return replyOf(request);
189             }
190           }
191           request.await();
192           continue;
193         }
194       } else {
195         // step 5 - reactivate request and insert into publist
196         activate(request);
197       }
198     }
199   }
200 
201   private void scanCombineApply() {
202     // traverse publist, combine ops & set "age" on requests
203     long now = System.currentTimeMillis();
204     Request current = publist;
205     boolean shutdown = this.shutdown;
206     while (current != null) {
207       WSlot op = current.requestOp;
208       if (op != null && current.response != null) {
209         throw new AssertionError(
210             "request " + op + " already have response: " + current.response);
211       }
212       if (current.deadlineIsPast(now)) {
213         replyTo(current, TIMEOUT);
214       } else if (op == CLAIM) {
215         // a claim request
216         // TODO optimize when claim comes before release on a depleted pool
217         if (current.isInterrupted()) {
218           replyTo(current, INTERRUPT);
219         } else if (shutdown) {
220           replyTo(current, SHUTDOWN);
221         } else if (liveStack != null) {
222           WSlot prospect = liveStack;
223           liveStack = prospect.next;
224           if (expired(prospect, now)) {
225             prospect.next = deadStack;
226             deadStack = prospect;
227             continue;
228           }
229           replyTo(current, prospect);
230         }
231       } else if (op == RELIEVE) {
232         // a relieve (reallocate) request
233         WSlot response = deadStack;
234         if (response != null) {
235           deadStack = response.next;
236           replyTo(current, response);
237         } else if (shutdown && liveStack != null) {
238           response = liveStack;
239           liveStack = response.next;
240           replyTo(current, response);
241         }
242       } else if (op != null) {
243         // a release request
244         if (expired(op, now)) {
245           op.next = deadStack;
246           deadStack = op;
247         } else {
248           op.next = liveStack;
249           liveStack = op;
250         }
251         replyTo(current, RELEASE);
252       }
253       current = current.next;
254     }
255   }
256 
257   private boolean expired(WSlot prospect, long now) {
258     return prospect.created + ttl < now;
259   }
260 
261   private void replyTo(Request request, WSlot response) {
262     request.requestOp = null;
263     request.response = response;
264     request.passCount = combiningPass;
265     request.hasTimeout = false;
266     request.unpark();
267   }
268 
269   private void cleanUp() {
270     // Called when the combiningPass count say it's time
271     Request current = publist;
272     // initial 'current' value is never null because publist at this point is
273     // guaranteed to contain at least one Request object - namely our own.
274     Request next = current.next;
275     while (next != null) {
276       if (expired(next) && next.requestOp == null) {
277         current.next = next.next;
278         next.deactivate();
279       } else {
280         current = next;
281         next = next.next;
282       }
283     }
284   }
285 
286   private boolean expired(Request request) {
287     return combiningPass - request.passCount > EXPIRE_PASS_COUNT;
288   }
289 
290   private WSlot replyOf(Request request) throws InterruptedException {
291     WSlot slot = request.response;
292     request.response = null;
293     if (slot == TIMEOUT) {
294       return null;
295     }
296     if (slot == INTERRUPT) {
297       Thread.interrupted();
298       throw new InterruptedException();
299     }
300     if (slot == SHUTDOWN) {
301       throw new IllegalStateException("pool is shutdown");
302     }
303     return slot;
304   }
305 
306   private void activate(Request request) {
307     request.active = true;
308     Request head;
309     do {
310       head = publist;
311       request.next = head;
312     } while (!publistCas.compareAndSet(this, head, request));
313   }
314 
315   public Completion shutdown() {
316     alloc.shutdown();
317     shutdown = true;
318     return alloc;
319   }
