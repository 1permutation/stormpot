16  package stormpot;

18  /**
19   * The expiration is used to determine if a given slot has expired, or
20   * otherwise become invalid.
21   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
22   */
23  public interface Expiration<T extends Poolable> {
24    /**
25     * Test whether the slot and poolable object, represented by the given
26     * {@link SlotInfo} object, is still valid, or if the pool should
27     * deallocate it and allocate a replacement.
28     * <p>
29     * If the method throws an exception, then that is taken to mean that the
30     * slot is invalid. How pools otherwise handle the exception - if it will
31     * bubble out, and if so, where - is implementation specific. For this
32     * reason, it is generally advised that Expirations do not throw
33     * exceptions.
34     * @param info An informative representative of the slot being tested.
35     * Never null.
36     * @return <code>true</code> if the slot and poolable in question should be
37     * deallocated, <code>false</code> if it is valid and elegible for claiming.
38     */
39    boolean hasExpired(SlotInfo<? extends T> info);
40  }
