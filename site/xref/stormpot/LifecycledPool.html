16  package stormpot;
17  
18  /**
19   * A LifecycledPool is a {@link Pool} that can be shut down to release any
20   * internal resources it might be holding on to.
21   * <p>
22   * Note that LifecycledPools are not guaranteed to have overwritten the
23   * {@link Object#finalize()} method. LifecycledPools are expected to rely on
24   * explicit clean-up for releasing their resources.
25   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
26   * @see Pool
27   * @param <T> The type of {@link Poolable} contained in this pool.
28   */
29  public interface LifecycledPool<T extends Poolable> extends Pool<T> {
30    /**
31     * Initiate the shut down process on this pool, and return a
32     * {@link Completion} instance representing the shut down procedure.
33     * <p>
34     * The shut down process is asynchronous, and the shutdown method is
35     * guaranteed not to wait for any claimed {@link Poolable Poolables} to
36     * be released.
37     * <p>
38     * The shut down process cannot complete before all Poolables are released
39     * back into the pool and {@link Allocator#deallocate(Poolable) deallocated},
40     * and all internal resources (such as threads, for instance) have been
41     * released as well. Only when all of these things have been taken care of,
42     * does the await methods of the Completion return.
43     * <p>
44     * Once the shut down process has been initiated, that is, as soon as this
45     * method is called, the pool can no longer be used and all calls to
46     * {@link #claim(Timeout)} will throw an {@link IllegalStateException}.
47     * Threads that are already waiting of objects in the claim method, will
48     * also wake up and receive an {@link IllegalStateException}.
49     * <p>
50     * All objects that are already claimed when this method is called,
51     * will continue to function until they are
52     * {@link Poolable#release() released}.
53     * <p>
54     * The shut down process is guaranteed to never deallocate objects that are
55     * currently claimed. Their deallocation will wait until they are released.
56     * @return A {@link Completion} instance that represents the shut down
57     * process.
58     */
59    Completion shutdown();
60  }
