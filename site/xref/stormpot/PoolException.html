16  package stormpot;

18  /**
19   * The PoolException may be thrown by a pool implementation in a number of
20   * circumstances:
21   * <ul>
22   * <li>If claim is called and the pool needs to
23   * {@link Allocator#allocate(Slot) allocate} a new object, but the allocation
24   * fails by returning <code>null</code> or throwing an exception.
25   * <li>Likewise if the {@link Reallocator#reallocate(Slot, Poolable)} method
26   * return <code>null</code> or throw an exception.
27   * <li>If the {@link Slot#release(Poolable)} method is misused, and the pool is
28   * able to detect this.
29   * <li>If the {@link Expiration#hasExpired(SlotInfo) expiration check} throws an
30   * exception.
31   * </ul>
32   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
33   */
34  public class PoolException extends RuntimeException {
35    private static final long serialVersionUID = -1908093409167496640L;
36  
37    /**
38     * Construct a new PoolException with the given message.
39     * @param message A description of the exception to be returned from
40     * {@link #getMessage()}.
41     * @see RuntimeException#RuntimeException(String)
42     */
43    public PoolException(String message) {
44      super(message);
45    }
46  
47    /**
48     * Construct a new PoolException with the given message and cause.
49     * @param message A description for the exception to be returned form
50     * {@link #getMessage()}.
51     * @param cause The underlying cause of this exception, as to be shown in the
52     * stack trace, and available through {@link #getCause()}.
53     * @see RuntimeException#RuntimeException(String, Throwable)
54     */
55    public PoolException(String message, Throwable cause) {
56      super(message, cause);
57    }
58  }
