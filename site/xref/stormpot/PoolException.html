16  package stormpot;

18  /**
19   * The PoolException may be thrown by a pool implementation in a number of
20   * circumstances:
21   * <ul>
22   * <li>If claim is called and the pool needs to
23   * {@link Allocator#allocate(Slot) allocate} a new object, but the allocation
24   * fails by returning <code>null</code> or throwing an exception.
25   * <li>If the {@link Slot#release(Poolable)} method is misused, and the pool is
26   * able to detect this.
27   * </ul>
28   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
29   *
30   */
31  public class PoolException extends RuntimeException {
32    private static final long serialVersionUID = -1908093409167496640L;
33  
34    /**
35     * Construct a new PoolException with the given message.
36     * @param message A description of the exception to be returned from
37     * {@link #getMessage()}.
38     * @see RuntimeException#RuntimeException(String)
39     */
40    public PoolException(String message) {
41      super(message);
42    }
43  
44    /**
45     * Construct a new PoolException with the given message and cause.
46     * @param message A description for the exception to be returned form
47     * {@link #getMessage()}.
48     * @param cause The underlying cause of this exception, as to be shown in the
49     * stack trace, and available through {@link #getCause()}.
50     * @see RuntimeException#RuntimeException(String, Throwable)
51     */
52    public PoolException(String message, Throwable cause) {
53      super(message, cause);
54    }
55  }
