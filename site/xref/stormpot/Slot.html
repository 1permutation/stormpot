16  package stormpot;
17  
18  /**
19   * A Slot represents a location in a Pool where objects can be allocated into,
20   * so that they can be claimed and released. The Slot is used by the allocated
21   * Poolables as a call-back, to tell the pool when an object is released.
22   * <p>
23   * The individual pool implementations provide their own Slot implementations.
24   * 
25   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
26   * @see Poolable
27   */
28  public interface Slot {
29    /**
30     * Signal to the pool that the currently claimed object at this slot, has been
31     * released.
32     * <p>
33     * It is a user error to release a slot that is not currently claimed. It is
34     * likewise a user error to release a Slot while inside the Allocators
35     * {@link Allocator#allocate(Slot) allocate} method, or the Expirations
36     * {@link Expiration#hasExpired(SlotInfo) hasExpired} method.
37     * <p>
38     * On the other hand, it is <em>not</em> an error to release a Poolable
39     * from a thread other than the one that claimed it.
40     * <p>
41     * Pools are free to throw a PoolException if they detect any of these
42     * wrong uses, but it is not guaranteed and the exact behaviour is not
43     * specified. "Unspecified behaviour" means that dead-locks and infinite
44     * loops are fair responses as well. Therefore, heed the advice and don't
45     * misuse release!
46     * <p>
47     * Pools must, however, guarantee that an object is never
48     * {@link Allocator#deallocate(Poolable) deallocated} more than once.
49     * This guarantee must hold even if release is misused.
50     * @param obj A reference to the Poolable object that is allocated for this
51     * slot, and is being released.
52     */
53    void release(Poolable obj);
54  }
