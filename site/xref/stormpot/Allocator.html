16  package stormpot;
17  
18  /**
19   * An Allocator is responsible for the creation and destruction of
20   * {@link Poolable} objects.
21   *
22   * This is where the objects in the Pool comes from. Clients of the Stormpot
23   * library needs to provide their own Allocator implementations.
24   *
25   * Implementations of this interface must be thread-safe, because there is no
26   * knowing whether pools will try to access it concurrently or not. The easiest
27   * way to achieve this is to just make the
28   * {@link Allocator#allocate(Slot) allocate}
29   * and {@link Allocator#deallocate(Poolable) deallocate} methods synchronised,
30   * but this might cause problems for the pools ability to allocate objects fast
31   * enough, if they are very slow to construct. Pools might want to deal with
32   * very slow allocations by allocating in more than one thread, but
33   * synchronizing the {@link #allocate(Slot)} method will render that strategy
34   * ineffective.
35   *
36   * A better approach to thread-safety is to not have any shared mutable state
37   * in the allocator, if at all possible.
38   * 
39   * @author Chris Vest <mr.chrisvest@gmail.com>
40   * @see stormpot.Reallocator
41   * @param <T> any type that implements Poolable.
42   */
43  public interface Allocator<T extends Poolable> {
44  
45    /**
46     * Create a fresh new instance of T for the given slot.
47     *
48     * The returned {@link Poolable} must obey the contract that, when
49     * {@link Poolable#release()} is called on it, it must delegate
50     * the call onto the {@link Slot#release(Poolable)} method of the here
51     * given slot object.
52     *
53     * Exceptions thrown by this method may propagate out through the
54     * {@link Pool#claim(Timeout) claim} method of a pool, in the form of being
55     * wrapped inside a {@link PoolException}. Pools must be able to handle these
56     * exceptions in a sane manner, and are guaranteed to return to a working
57     * state if an Allocator stops throwing exceptions from its allocate method.
58     * @param slot The slot the pool wish to allocate an object for.
59     * Implementers do not need to concern themselves with the details of a
60     * pools slot objects. They just have to call release on them as the
61     * protocol demands.
62     * @return A newly created instance of T. Never `null`.
63     * @throws Exception If the allocation fails.
64     */
65    T allocate(Slot slot) throws Exception;
66  
67    /**
68     * Deallocate, if applicable, the given Poolable and free any resources
69     * associated with it.
70     *
71     * This is an opportunity to close any connections or files, flush buffers,
72     * empty caches or what ever might need to be done to completely free any
73     * resources represented by this Poolable.
74     *
75     * Note that a Poolable must never touch its slot object after it has been
76     * deallocated.
77     *
78     * Pools, on the other hand, will guarantee that the same object is never
79     * deallocated more than once.
80     *
81     * Note that pools will always silently swallow exceptions thrown by the
82     * deallocate method. They do this because there is no knowing whether the
83     * deallocation of an object will be done synchronously by a thread calling
84     * {@link Poolable#release() release} on a Poolable, or asynchronously by
85     * a clean-up thread inside the pool.
86     *
87     * Deallocation from the release of an expired object, and deallocation from
88     * the shut down procedure of a {@link LifecycledPool} behave the same way
89     * in this regard. They will both silently swallow any exception thrown.
90     *
91     * On the other hand, pools are guaranteed to otherwise correctly deal with
92     * any exception that might be thrown. The shut down procedure will still
93     * complete, and release will still maintain the internal data structures of
94     * the pool to make the slot available for new allocations.
95     *
96     * If you need to somehow specially deal with the exceptions thrown by the
97     * deallocation of objects, then you should do this in the allocator itself,
98     * or in a wrapper around your allocator.
99     * @param poolable The non-null Poolable instance to be deallocated.
100    * @throws Exception if the deallocation encounters an error.
101    */
102   void deallocate(T poolable) throws Exception;
103 }
