29  /**
30   * BlazePool is a highly optimised {@link LifecycledResizablePool}
31   * implementation that consists of a queues of Poolable instances, the access
32   * to which is made faster with clever use of ThreadLocals.
33   * <p>
34   * Object allocation always happens in a dedicated thread, off-loading the 
35   * cost of allocating the pooled objects. This should lead to reduced deviation
36   * in the times it takes claim method to complete, provided the pool is not
37   * depleted.
38   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
39   * @param <T> The type of {@link Poolable} managed by this pool.
40   */
41  public final class BlazePool<T extends Poolable>
42  implements LifecycledResizablePool<T> {
43    private final BlockingQueue<BSlot<T>> live;
44    private final BlockingQueue<BSlot<T>> dead;
45    private final BAllocThread<T> allocThread;
46    private final Expiration<? super T> deallocRule;
47    // TODO consider making it a ThreadLocal of Ref<QSlot>, hoping that maybe
48    // writes to it will be faster in not requiring a ThreadLocal look-up.
49    private final ThreadLocal<BSlot<T>> tlr;
50    private volatile boolean shutdown = false;
51    
52    /**
53     * Construct a new BlazePool instance based on the given {@link Config}.
54     * @param config The pool configuration to use.
55     */
56    public BlazePool(Config<T> config) {
57      live = new LinkedBlockingQueue<BSlot<T>>();
58      dead = new LinkedBlockingQueue<BSlot<T>>();
59      tlr = new ThreadLocal<BSlot<T>>();
60      synchronized (config) {
61        config.validate();
62        allocThread = new BAllocThread<T>(live, dead, config);
63        deallocRule = config.getExpiration();
64      }
65      allocThread.start();
66    }

68    public T claim(Timeout timeout) throws PoolException,
69        InterruptedException {
70      if (timeout == null) {
71        throw new IllegalArgumentException("timeout cannot be null");
72      }
73      BSlot<T> slot = tlr.get();
74      // Note that the TLR slot at this point might have been tried by another
75      // thread, found to be expired, put on the dead-queue and deallocated.
76      // We handle this because slots always transition to the dead state before
77      // they are put on the dead-queue, and if they are dead, then the
78      // slot.live2claimTlr() call will fail.
79      // Then we will eventually find another slot from the live-queue that we
80      // can claim and make our new TLR slot.
81      if (slot != null && slot.live2claimTlr()) {
82        // Attempt the claim before checking the validity, because we might
83        // already have claimed it.
84        // If we checked validity before claiming, then we might find that it
85        // had expired, and throw it in the dead queue, causing a claimed
86        // Poolable to be deallocated before it is released.
87        if (!isInvalid(slot)) {
88          slot.incrementClaims();
89          return slot.obj;
90        }
91        // We managed to tlr-claim the slot, but it turned out to be no good.
92        // That means we now have to transition it from tlr-claimed to dead.
93        // However, since we didn't pull it off of the live-queue, it might still
94        // be in the live-queue. And since it might be in the live-queue, it
95        // can't be put on the dead-queue. And since it can't be put on the
96        // dead-queue, it also cannot transition to the dead state.
97        // This effectively means that we have to transition it back to the live
98        // state, and then let some pull it off of the live-queue, check it
99        // again, and only then put it on the dead-queue.
100       // It's cumbersome, but we have to do it this way, in order to prevent
101       // duplicate entries in the queues. Otherwise we'd have a nasty memory
102       // leak on our hands.
103     }
104     long deadline = timeout.getDeadline();
105     boolean notClaimed = true;
106     do {
107       long timeoutLeft = timeout.getTimeLeft(deadline);
108       slot = live.poll(timeoutLeft, timeout.getBaseUnit());
109       if (slot == null) {
110         // we timed out while taking from the queue - just return null
111         return null;
112       }
113       // Again, attempt to claim before checking validity. We mustn't kill
114       // objects that are already claimed by someone else.
115       do {
116         // We pulled a slot off the queue. If we can transition it to the
117         // claimed state, then it means it wasn't tlr-claimed and we got it.
118         // Note that the slot at this point be in any queue.
119         notClaimed = !slot.live2claim();
120         // If we fail to claim it, then it means that it is tlr-claimed by
121         // someone else. We know this because slots in the live-queue can only
122         // be either live or tlr-claimed. There is no transition to claimed
123         // or dead without first pulling the slot off the live-queue.
124         // Note that the slot at this point is not in any queue, and we can't
125         // put it back into the live-queue, because that could lead to
126         // busy-looping on tlr-claimed slots which would waste CPU cycles when
127         // the pool is depleted. We must instead make sure that tlr-claimer
128         // transition to a proper claimer, such that he will make sure to
129         // release the slot back into the live-queue once he is done with it.
130         // However, as we are contemplating this, he might have already
131         // released it again, which means that it is live and we can't make our
132         // transition because it is now too late for him to put it back into
133         // the live-queue. On the other hand, if the slot is now live, it means
134         // that we can claim it for our selves. So we loop on this.
135       } while (notClaimed && !slot.claimTlr2claim());
136       // If we could not live->claimed but tlr-claimed->claimed, then
137       // we mustn't check isInvalid, because that might send it to the
138       // dead-queue *while somebody else thinks they've TLR-claimed it!*
139       // We handle this in the outer loop: if we couldn't claim, then we retry
140       // the loop.
141     } while (notClaimed || isInvalid(slot));
142     slot.incrementClaims();
143     tlr.set(slot);
144     return slot.obj;
145   }

147   private boolean isInvalid(BSlot<T> slot) {
148     checkForPoison(slot);
149     boolean invalid = true;
150     RuntimeException exception = null;
151     try {
152       invalid = deallocRule.hasExpired(slot);
153     } catch (RuntimeException ex) {
154       exception = ex;
155     }
156     if (invalid) {
157       // it's invalid - into the dead queue with it and continue looping
158       kill(slot);
159       if (exception != null) {
160         throw exception;
161       }
162     }
163     return invalid;
164   }
165 
166   private void checkForPoison(BSlot<T> slot) {
167     if (slot == allocThread.POISON_PILL) {
168       // The poison pill means the pool has been shut down. The pill was
169       // transitioned from live to claimed just prior to this check, so we
170       // must transition it back to live and put it back into the live-queue
171       // before throwing our exception.
172       // Because we always throw when we see it, it will never become a
173       // tlr-slot, and so we don't need to worry about transitioning from
174       // tlr-claimed to live.
175       slot.claim2live();
176       live.offer(allocThread.POISON_PILL);
177       throw new IllegalStateException("pool is shut down");
178     }
179     if (slot.poison != null) {
180       Exception poison = slot.poison;
181       kill(slot);
182       throw new PoolException("allocation failed", poison);
183     }
184     if (shutdown) {
185       kill(slot);
186       throw new IllegalStateException("pool is shut down");
187     }
188   }
189 
190   private void kill(BSlot<T> slot) {
191     // The use of claim2dead() here ensures that we don't put slots into the
192     // dead-queue more than once. Many threads might have this as their
193     // TLR-slot and try to tlr-claim it, but only when a slot has been normally
194     // claimed, that is, pulled off the live-queue, can it be put into the
195     // dead-queue. This helps ensure that a slot will only ever be in at most
196     // one queue.
197     for (;;) {
198       int state = slot.getState();
199       if (state == BSlot.CLAIMED && slot.claim2dead()) {
200         dead.offer(slot);
201         return;
202       }
203       if (state == BSlot.TLR_CLAIMED && slot.claimTlr2live()) {
204         return;
205       }
206     }
207   }

209   public Completion shutdown() {
210     shutdown = true;
211     allocThread.interrupt();
212     return new BPoolShutdownCompletion(allocThread);
213   }

215   public void setTargetSize(int size) {
216     if (size < 1) {
217       throw new IllegalArgumentException("target size must be at least 1");
218     }
219     allocThread.setTargetSize(size);
220   }
