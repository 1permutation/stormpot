16  package stormpot;
17  
18  import java.util.concurrent.TimeUnit;
19  
20  /**
21   * A Timeout represents the maximum amount of time a caller is willing to wait
22   * for a blocking operation to complete.
23   *
24   * Timeouts are independent of their units, so two timeouts of equivalent
25   * duration but constructed in different units, will be equal to each other and
26   * work exactly the same.
27   *
28   * Timeouts are also independent of "calendar time" in the sense that they
29   * represent and work as a duration of absolute time. In other words, timeouts
30   * do not grow or shrink with passing leap seconds or daylight savings time
31   * adjustments.
32   *
33   * @author Chris Vest <mr.chrisvest@gmail.com>
34   */
35  // TODO 3.0 make class final
36  public class Timeout {
37    private final long timeout;
38    private final TimeUnit unit;
39    private final long timeoutBase;
40    
41    /**
42     * Construct a new timeout with the given value and unit. The unit cannot be
43     * `null`, but the timeout value is unrestricted. The meaning of a
44     * negative timeout value is specific to the implementation of the use site,
45     * but typically means that no amount of blocking or waiting is tolerated.
46     * @param timeout A numerical value for the timeout. Can be zero or negative,
47     * though the meaning is implementation specific.
48     * @param unit The unit of the timeout value. Never `null`.
49     */
50    public Timeout(long timeout, TimeUnit unit) {
51      if (unit == null) {
52        throw new IllegalArgumentException("The TimeUnit cannot be null");
53      }
54      this.timeout = timeout;
55      this.unit = unit;
56      this.timeoutBase = getBaseUnit().convert(timeout, unit);
57    }
58  
59    /**
60     * Get the timeout value in terms of the {@link #getUnit() unit}.
61     * @return A numerical value of the timeout. Possibly zero or negative.
62     */
63    public long getTimeout() {
64      return timeout;
65    }
66  
67    /**
68     * Get the unit for the {@link #getTimeout() timeout value}.
69     * @return The {@link TimeUnit} of the timeout value. Never `null`.
70     */
71    public TimeUnit getUnit() {
72      return unit;
73    }
74  
75    /**
76     * Calculate a deadline, as an instant in the future, in terms of the
77     * {@link #getBaseUnit() base unit}. Once you have a deadline, you can ask
78     * how much time is left until it transpires, with the
79     * {@link #getTimeLeft(long)} method, giving the deadline as an argument.
80     *
81     * If the {@link #getTimeout() timeout value} is really small, zero or
82     * negative, then the deadline might be an instant in the past.
83     * @return A numerical value that represents the deadline from "now" until
84     * this timeout has passed.
85     */
86    public long getDeadline() {
87      return now() + timeoutBase;
88    }
89  
90    /**
91     * Get the timeout value in terms of the {@link #getBaseUnit() base unit}.
92     * @return A numerical value of the timeout. Possibly zero or negative.
93     */
94    public long getTimeoutInBaseUnit() {
95      return timeoutBase;
96    }
97  
98    /**
99     * Calculate the amount of time that is left until the deadline, in terms
100    * of the {@link #getBaseUnit() base unit}. The argument is a deadline value
101    * that has been calculated with the {@link #getDeadline()} method.
102    * @param deadline The reference deadline from {@link #getDeadline()}.
103    * @return The amount of time, in terms of the
104    * {@link #getBaseUnit() base unit}, that is left until the deadline. If
105    * this value is negative, then the deadline has transpired.
106    */
107   public long getTimeLeft(long deadline) {
108     return deadline - now();
109   }
110 
111   /**
112    * Get the unit of precision for the underlying clock, that is used by
113    * {@link #getDeadline()} and {@link #getTimeLeft(long)}.
114    * @return TimeUnit The unit of precision used by the clock in this Timeout.
115    */
116   public TimeUnit getBaseUnit() {
117     return TimeUnit.NANOSECONDS;
118   }
119 
120   private long now() {
121     return System.nanoTime();
122   }
123 
124   @Override
125   public int hashCode() {
126     return 31 * (1 + (int) (timeoutBase ^ (timeoutBase >>> 32)));
127   }
128 
129   /**
130    * Timeouts of equivalent duration are equal, even if they were constructed
131    * with different units.
132    * @return `true` if this Timeout value is equal to the given
133    * Timeout value, `false` otherwise.
134    */
135   @Override
136   public boolean equals(Object obj) {
137     if (!(obj instanceof Timeout)) {
138       return false;
139     }
140     Timeout that = (Timeout) obj;
141     return this.timeoutBase == that.timeoutBase;
142   }
143 }
