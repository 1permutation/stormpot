34    public BSlot(BlockingQueue<BSlot<T>> live, AtomicInteger poisonedSlots) {
35      // Volatile write in the constructor: This object must be safely published,
36      // so that we are sure that the volatile write happens-before other
37      // threads observe the pointer to this object.
38      super(DEAD, live, poisonedSlots);
39    }
40    
41    public void release(Poolable obj) {
42      int slotState = getClaimState();
43      lazySet(LIVING);
44      if (slotState == CLAIMED) {
45        live.offer(this);
46      }
47    }
48  
49    private int getClaimState() {
50      int slotState = get();
51      if (slotState > TLR_CLAIMED) {
52        throw badStateOnTransitionToLive(slotState);
53      }
54      return slotState;
55    }
56  
57    private PoolException badStateOnTransitionToLive(int slotState) {
58      String state;
59      switch (slotState) {
60        case DEAD: state = "DEAD"; break;
61        case LIVING: state = "LIVING"; break;
62        default: state = "STATE[" + slotState + "]";
63      }
64      return new PoolException("Slot release from bad state: " + state + ". " +
65          "You most likely called release() twice on the same object.");
66    }
