16  package stormpot.qpool;
17  
18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.CountDownLatch;
20  import java.util.concurrent.TimeUnit;
21  import java.util.concurrent.locks.LockSupport;
22  
23  import stormpot.Allocator;
24  import stormpot.Config;
25  import stormpot.Poolable;
26  import stormpot.Timeout;
27  
28  class QAllocThread<T extends Poolable> extends Thread {
29    /**
30     * Special slot used to signal that the pool has been shut down.
31     */
32    final QSlot<T> POISON_PILL = new QSlot<T>(null);
33    
34    private final CountDownLatch completionLatch;
35    private final BlockingQueue<QSlot<T>> live;
36    private final BlockingQueue<QSlot<T>> dead;
37    private final Allocator<T> allocator;
38    private volatile int targetSize;
39    private int size;
40    private long deadPollTimeout = 1;
41  
42    public QAllocThread(
43        BlockingQueue<QSlot<T>> live, BlockingQueue<QSlot<T>> dead,
44        Config<T> config) {
45      this.targetSize = config.getSize();
46      completionLatch = new CountDownLatch(1);
47      this.allocator = config.getAllocator();
48      this.size = 0;
49      this.live = live;
50      this.dead = dead;
51    }
52  
53    @Override
54    public void run() {
55      continuouslyReplenishPool();
56      shutPoolDown();
57      completionLatch.countDown();
58    }
59  
60    private void continuouslyReplenishPool() {
61      try {
62        for (;;) {
63          deadPollTimeout = size == targetSize? 50 : 1;
64          if (size < targetSize) {
65            QSlot<T> slot = new QSlot<T>(live);
66            alloc(slot);
67          }
68          QSlot<T> slot = dead.poll(deadPollTimeout, TimeUnit.MILLISECONDS);
69          if (size > targetSize) {
70            slot = slot == null? live.poll() : slot;
71            if (slot != null) {
72              dealloc(slot);
73            }
74          } else if (slot != null) {
75            dealloc(slot);
76            alloc(slot);
77            // mutation testing might note that the above alloc() call can be
78            // removed... that's okay, it's really just an optimisation that
79            // prevents us from creating new slots all the time - we reuse them.
80          }
81        }
82      } catch (InterruptedException _) {
83        // this means we've been shut down.
84        // let the poison-pill enter the system
85        live.offer(POISON_PILL);
86      }
87    }
88  
89    private void shutPoolDown() {
90      while (size > 0) {
91        QSlot<T> slot = dead.poll();
92        if (slot == null) {
93          slot = live.poll();
94        }
95        if (slot == POISON_PILL) {
96          // FindBugs complains that we ignore a possible exceptional return
97          // value from offer(). However, since the queues are unbounded, an
98          // offer will never fail.
99          live.offer(POISON_PILL);
100         slot = null;
101       }
102       if (slot == null) {
103         LockSupport.parkNanos(10000000); // 10 millis
104       } else {
105         dealloc(slot);
106       }
107     }
108   }
109 
110   private void alloc(QSlot<T> slot) {
111     try {
112       slot.obj = allocator.allocate(slot);
113       if (slot.obj == null) {
114         slot.poison = new NullPointerException("allocation returned null");
115       }
116     } catch (Exception e) {
117       slot.poison = e;
118     }
119     size++;
120     slot.created = System.currentTimeMillis();
121     slot.claims = 0;
122     slot.claimed.set(true);
123     slot.release(slot.obj);
124   }
125 
126   private void dealloc(QSlot<T> slot) {
127     size--;
128     try {
129       if (slot.poison == null) {
130         allocator.deallocate(slot.obj);
131       }
132     } catch (Exception _) { // NOPMD
133       // ignored as per specification
134     }
135     slot.poison = null;
136     slot.obj = null;
137   }
138 
139   boolean await(Timeout timeout) throws InterruptedException {
140     return completionLatch.await(timeout.getTimeout(), timeout.getUnit());
141   }
142 
143   void setTargetSize(int size) {
144     this.targetSize = size;
145     LockSupport.unpark(this);
146   }
147 
148   int getTargetSize() {
149     return targetSize;
150   }
151 }
