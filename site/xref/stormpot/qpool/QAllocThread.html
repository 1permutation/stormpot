16  package stormpot.qpool;
17  
18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.CountDownLatch;
20  import java.util.concurrent.TimeUnit;
21  import java.util.concurrent.locks.LockSupport;
22  
23  import stormpot.Allocator;
24  import stormpot.Config;
25  import stormpot.Poolable;
26  
27  @SuppressWarnings("unchecked")
28  class QAllocThread<T extends Poolable> extends Thread {
29    private final CountDownLatch completionLatch;
30    private final BlockingQueue<QSlot<T>> live;
31    private final BlockingQueue<QSlot<T>> dead;
32    private final Allocator<T> allocator;
33    private final int targetSize;
34    private final long ttlMillis;
35    private int size;
36    private long deadPollTimeout = 1;
37  
38    public QAllocThread(
39        BlockingQueue<QSlot<T>> live, BlockingQueue<QSlot<T>> dead,
40        Config<T> config) {
41      this.targetSize = config.getSize();
42      completionLatch = new CountDownLatch(1);
43      this.allocator = config.getAllocator();
44      this.size = 0;
45      this.live = live;
46      this.dead = dead;
47      ttlMillis = config.getTTLUnit().toMillis(config.getTTL());
48    }
49  
50    @Override
51    public void run() {
52      try {
53        for (;;) {
54          if (size < targetSize) {
55            QSlot slot = new QSlot(live);
56            alloc(slot);
57            if (size == targetSize) {
58              deadPollTimeout = 50;
59            }
60          }
61          QSlot slot = dead.poll(deadPollTimeout, TimeUnit.MILLISECONDS);
62          if (slot != null) {
63            dealloc(slot);
64            alloc(slot);
65          }
66        }
67      } catch (InterruptedException e) {
68        // we're shut down
69        while (size > 0) {
70          QSlot<T> slot = dead.poll();
71          if (slot == null) {
72            slot = live.poll();
73          }
74          if (slot == QueuePool.KILL_PILL) {
75            // FindBugs complains that we ignore a possible exceptional return
76            // value from offer(). However, since the queues are unbounded, an
77            // offer will never fail.
78            live.offer(QueuePool.KILL_PILL);
79            slot = null;
80          }
81          if (slot == null) {
82            LockSupport.parkNanos(10000000); // 10 millis
83          } else {
84            dealloc(slot);
85          }
86        }
87      } finally {
88        completionLatch.countDown();
89      }
90    }
91  
92    private void alloc(QSlot slot) {
93      try {
94        slot.obj = allocator.allocate(slot);
95        if (slot.obj == null) {
96          slot.poison = new NullPointerException("allocation returned null");
97        }
98      } catch (Exception e) {
99        slot.poison = e;
100     }
101     size++;
102     slot.expires = System.currentTimeMillis() + ttlMillis;
103     slot.claim();
104     slot.release(slot.obj);
105   }
106 
107   private void dealloc(QSlot<T> slot) {
108     size--;
109     try {
110       if (slot.poison == null) {
111         allocator.deallocate(slot.obj);
112       }
113     } catch (Exception _) {
114       // ignored as per specification
115     } finally {
116       slot.poison = null;
117       slot.obj = null;
118     }
119   }
120 
121   public void await() throws InterruptedException {
122     completionLatch.await();
123   }
124 
125   public boolean await(long timeout, TimeUnit unit)
126       throws InterruptedException {
127     return completionLatch.await(timeout, unit);
128   }
129 }
